WITH cte AS (
	SELECT person_name , turn,
    SUM(Weight) OVER (ORDER BY Turn) AS x
	FROM Queue 
	)
SELECT person_name 
FROM cte
WHERE x <= 1000
ORDER BY turn DESC
LIMIT 1
